async function createSoapClientAsync(context: IContext) {

  let url =
    process.env.ENVIRONTMENT == "development"
      ? process.env.MONEYGRAM_DEV_URL
      : process.env.MONEYGRAM_PROD_URL;

  let client = await soap.createClientAsync(url);

  client.on("request", (xml: any) => {
    Logger.info("Request: " + xml);
    Logger.info("cod_empresa: " + context.cod_empresa);
    Logger.info("clave_cobro: " + context.referenceNumber);
    guardar_bitacora_mg("REQUEST", xml, context);
  });

  client.on("response", (xml: any) => {
    Logger.info("Response: " + xml);
    Logger.info("cod_empresa: " + context.cod_empresa);
    Logger.info("clave_cobro: " + context.referenceNumber);
    guardar_bitacora_mg("RESPONSE", xml, context);
  });

  client.on("soapError", (err: any, eid: string) => {
    Logger.info("cod_empresa: " + context.cod_empresa);
    Logger.info("clave_cobro: " + context.referenceNumber);
    Logger.error(
      "SOAP Error: " +
        err.toString().replaceAll("Error: soapenv:Server: undefined: ", "")
    );
  });

  if (client == null) {
    Logger.error("Error creating SOAP client");
    Logger.error(client);
    throw new Error("Error creating SOAP client");
  }
  return client;
}

    if(error.toString().includes("Error: soapenv:Server: undefined: ")){

/**
 * Function to remove undefined properties from an object
 * 
 * @param objeto Object to be cleared of undefined properties
 * @returns void
 */
export function clearUndefinedProperties(objeto: any):void {
    for (var i = 0, keys = Object.keys(objeto), len = keys.length; i < len; i++) {
        if (typeof objeto[keys[i]] === "undefined") {
            delete objeto[keys[i]];
        }
    }
}
