/**
 * Realiza la búsqueda de un Cash Pick Up Order en Dandelion
 * @param {object} requestBody - El body JSON recibido del request (Express)
 * @param {object} context - El contexto armado (cod_empresa, clave_cobro, endpoint, etc)
 * @returns {Promise<object>} - Respuesta de Dandelion
 */
async function searchCashPickUpOrder(requestBody, context) {
    // 1. Armamos los headers requeridos por Dandelion
    const headers = {
        'Content-Type': 'application/json',                   // Tipo de contenido JSON
        'ria-CallerCorrelationId': requestBody.correlationId  // ID único para trazar la petición, lo puede generar front o aquí (ej: UUID)
            || Date.now().toString(),
        'ria-CallDateTimeLocal': requestBody.callDateTimeLocal // Fecha/hora local de la llamada, formato YYYYMMDDHHmmss
            || DateTimeStamp.nowYYYYMMDDHHMMSS(),
        'ria-AgentId': requestBody.agentId,                   // ID de agente proporcionado por Ria/Dandelion
        'Ocp-Apim-Subscription-Key': process.env.DANDELION_KEY, // Clave API, tomada de variables de entorno
        // Puedes incluir los opcionales: UserId, DeviceId, DeviceType si tienes esa info
    };

    // 2. Armamos el body JSON según la documentación
    const body = {
        BeneAmount: requestBody.BeneAmount,                   // Monto a entregar al beneficiario (requerido)
        CorrespLocID: requestBody.CorrespLocID,               // Payout location identifier (opcional)
        DateTimeLocal: requestBody.DateTimeLocal              // Fecha/hora local (YYYYMMDDHHmmss, requerido)
            || DateTimeStamp.nowYYYYMMDDHHMMSS(),
        DateTimeUTC: requestBody.DateTimeUTC                  // Fecha/hora UTC (YYYYMMDDHHmmss, requerido)
            || DateTimeStamp.nowUTCYYYYMMDDHHmmss(),
        PIN: requestBody.PIN                                  // PIN de la orden (requerido)
    };

    // 3. Guardamos la bitácora del REQUEST
    await guardar_bitacora_dandelion('REQUEST', { headers, body }, context);

    // 4. Llamamos al endpoint REST de Dandelion usando el cliente HTTP
    let response;
    try {
        response = await sendHttpRequest(
            'POST',
            'https://mt-staging.dandelionpayments.com/v1/cash_pick_up/search',
            headers,
            body
        );
    } catch (error) {
        // 5. Si hubo error, lo guardamos y lo relanzamos
        await guardar_bitacora_dandelion('ERROR', error, context);
        Logger.errorJSON(error);
        throw error;
    }

    // 6. Guardamos la bitácora del RESPONSE
    await guardar_bitacora_dandelion('RESPONSE', response, context);

    // 7. Retornamos la respuesta de Dandelion a Express
    return response;
}

const axios = require('axios');

/**
 * Función para enviar solicitudes HTTP REST a Dandelion
 * @param {string} method - El método HTTP (por ejemplo, 'POST')
 * @param {string} url - La URL del endpoint de Dandelion
 * @param {object} headers - Un objeto con los headers requeridos
 * @param {object} body - El cuerpo del request (JSON)
 * @returns {Promise<object>} - La respuesta JSON de Dandelion
 */
async function sendHttpRequest(method, url, headers, body) {
    try {
        Logger.info(`HTTP ${method} ${url}`);   // Log: mostramos método y URL
        Logger.infoJSON({ headers, body });     // Log: mostramos método y URL
         const response = await axios({           // Hacemos la llamada HTTP
            method,                             // Método HTTP ('POST', 'GET', etc)
            url,                                // URL del endpoint
            headers,                            // Headers, incluyendo autenticación y contexto
            data: body                          // Body de la petición, en formato JSON
        });

        Logger.infoJSON(response.data);         // Log: respuesta recibida de Dandelion
        return response.data;                   // Retornamos la data (json) al que llamó esta función
    } catch (error) {
        // Si hay error, loggeamos el error (respuesta de Dandelion o error de red)
        Logger.errorJSON(error.response ? error.response.data : error);
        throw error;                            // Relanzamos para manejo superior
    }
}

const { DateTime } = require('luxon');

/**
 * Clase utilitaria para manejo de fechas y horas en formato requerido por Dandelion (YYYYMMDDHHmmss)
 */
class DateTimeStamp {
    /**
     * Retorna la fecha/hora actual en formato YYYYMMDDHHmmss (tiempo local)
     * @returns {string}
     */
    static nowYYYYMMDDHHMMSS() {
        // DateTime.now() obtiene la fecha/hora actual (local)
        // .toFormat('yyyyLLddHHmmss') la convierte a string con formato requerido
        return DateTime.now().toFormat('yyyyLLddHHmmss');
    }

    /**
     * Retorna la fecha/hora actual en formato YYYYMMDDHHmmss (UTC)
     * @returns {string}
     */
    static nowUTCYYYYMMDDHHMMSS() {
        // DateTime.utc() obtiene la fecha/hora en UTC
        // .toFormat(...) lo convierte al formato string requerido
        return DateTime.utc().toFormat('yyyyLLddHHmmss');
    }

    /**
     * Convierte una fecha JS a formato YYYYMMDDHHmmss
     * @param {Date|string|DateTime} date - La fecha a convertir
     * @returns {string}
     */
    static toYYYYMMDDHHMMSS(date) {
        // Si es string, intentamos parsear como ISO
        let dt = typeof date === 'string'
            ? DateTime.fromISO(date)
            : DateTime.fromJSDate(date);

        // Si ya es DateTime, no lo transformamos más
        if (date instanceof DateTime) {
            dt = date;
        }

        return dt.toFormat('yyyyLLddHHmmss');
    }
}

module.exports = { DateTimeStamp };

/**
 * Maneja y adapta los errores provenientes del API Dandelion (HTTP/REST).
 * Devuelve mensajes claros para el cliente, extraídos del body de error.
 * @param {Error} error - El error capturado en un try/catch (axios)
 * @returns {object} - Un objeto de error listo para enviar como respuesta HTTP
 */
function handleDandelionError(error) {
    // Si es un error de axios y la respuesta trae data, intentamos formatearla
    if (error.response && error.response.data) {
        const data = error.response.data;

        // Si Dandelion envía el error en la estructura { Errors: { ErrorMsg: "...", ... } }
        if (data.Errors && data.Errors.ErrorMsg) {
            return {
                status: 'error',
                message: data.Errors.ErrorMsg,
                details: data.Errors.InputParams || null
            };
        }

        // Si Dandelion envía otros mensajes de error, los retornamos todos
        return {
            status: 'error',
            message: data.message || 'Ocurrió un error en la comunicación con Dandelion.',
            raw: data
        };
    }

    // Si el error es un string plano
    if (typeof error === 'string') {
        return {
            status: 'error',
            message: error
        };
    }

    // Si el error es un objeto plano con mensaje
    if (typeof error === 'object' && error.message) {
        return {
            status: 'error',
            message: error.message
        };
    }

    // Último recurso: error desconocido
    return {
        status: 'error',
        message: 'Ocurrió un error desconocido en el proxy Dandelion.'
    };
}

module.exports = { handleDandelionError };

/**
 * Extrae el contexto del request HTTP.
 * Este contexto se utiliza para bitácora y trazabilidad, igual que en el proxy MoneyGram.
 * @param {object} req - El objeto request de Express
 * @returns {object} context - Objeto con cod_empresa, clave_cobro y endpoint
 */
function getContext(req) {
    // 1. Extraemos cod_empresa del body (si no está, default a 1)
    const cod_empresa = req.body.cod_empresa || 1;

    // 2. Extraemos clave_cobro del body (si no está, default a 2)
    const clave_cobro = req.body.clave_cobro || 2;

    // 3. Extraemos el endpoint invocado (de la URL del request)
    // Si _parsedUrl no está, usamos 'N/A'
    const endpoint = (req._parsedUrl && req._parsedUrl.path) || 'N/A';

    // 4. Armamos el objeto contexto
    const context = {
        cod_empresa,
        clave_cobro,
        endpoint
    };

    // 5. Retornamos el contexto armado
    return context;
}

module.exports = { getContext };
